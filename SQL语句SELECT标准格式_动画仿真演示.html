📋 SELECT 语句标准格式
R SELECT [DISTINCT] column1, column2, ... # 选择列
R FROM table_name # 数据来源
O [WHERE condition] # 筛选条件
O [GROUP BY column] # 分组
O [HAVING condition] # 分组后筛选
O [ORDER BY column [ASC|DESC]] # 排序
O [LIMIT number] # 限制结果数量
