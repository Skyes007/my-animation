📋 ALTER TABLE 标准格式
R ALTER TABLE table_name
O ADD [COLUMN] column_name column_definition [FIRST|AFTER existing_column], # 添加列（可选）
O DROP [COLUMN] column_name, # 删除列（可选）
O MODIFY [COLUMN] column_name column_definition, # 修改列定义（可选）
O CHANGE [COLUMN] old_column_name new_column_name column_definition, # 重命名列（可选）
O RENAME [TO|AS] new_table_name, # 重命名表（可选）
O ADD PRIMARY KEY (column_name), # 添加主键（可选）
O DROP PRIMARY KEY, # 删除主键（可选）
O ADD FOREIGN KEY (column_name) REFERENCES other_table(column_name), # 添加外键（可选）
O DROP FOREIGN KEY constraint_name, # 删除外键（可选）
O ADD INDEX index_name (column_name), # 添加索引（可选）
O DROP INDEX index_name, # 删除索引（可选）
O ADD CONSTRAINT constraint_name CHECK (condition), # 添加检查约束（可选）
O DROP CONSTRAINT constraint_name; # 删除约束（可选）
