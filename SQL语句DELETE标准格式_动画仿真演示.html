📋 DELETE 语句标准格式
R DELETE FROM table_name # 指定要删除数据的表
O WHERE condition; # 指定删除条件（可选）
# 常用条件运算符：=, <>, >, <, >=, <=, AND, OR, IN, LIKE
# 示例：
DELETE FROM students WHERE id = 1;
DELETE FROM students WHERE age > 20;
DELETE FROM students WHERE name LIKE '张%';
