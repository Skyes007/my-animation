📋 UPDATE 语句标准格式
R UPDATE table_name
R SET column1 = value1, column2 = value2, ...
O WHERE condition; # 条件筛选（可选）
R UPDATE table_name
R SET column1 = value1, column2 = value2, ...
O WHERE column_name IN (value1, value2, ...); # IN条件
O WHERE column_name BETWEEN value1 AND value2; # BETWEEN条件
O WHERE column_name LIKE pattern; # LIKE条件
