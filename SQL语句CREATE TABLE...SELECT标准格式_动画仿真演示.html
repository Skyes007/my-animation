📋 CREATE TABLE...SELECT 语句标准格式
R CREATE TABLE table_name # 创建新表
O [(column_definition, ...)] # 可选的列定义
R AS # 指定数据来源
R SELECT ... FROM ... # 选择查询
O [WHERE ...] # 可选的筛选条件
O [GROUP BY ...] # 可选的聚合
O [ORDER BY ...] # 可选的排序
O [LIMIT ...] # 可选的限制
