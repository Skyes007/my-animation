📋 CREATE TABLE 标准格式
R CREATE TABLE table_name
R (
R column1 datatype [constraints], # 列定义（必需）
O [UNIQUE (column_name)], # 唯一约束（可选）
O [DEFAULT default_value], # 默认值（可选）
O [CONSTRAINT constraint_name] PRIMARY KEY (column1), # 主键约束（可选）
O FOREIGN KEY (column2) REFERENCES other_table(column), # 外键约束（可选）
O CHECK (condition) # 检查约束（可选）
R );
O [ENGINE = engine_name]; # 存储引擎（可选）
